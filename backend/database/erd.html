    users {
        UUID id PK
        VARCHAR email UK
        VARCHAR password_hash
        VARCHAR full_name
        VARCHAR phone
        TEXT avatar_url
        ENUM role "super_admin|admin|staff|user"
        BOOLEAN is_active
        BOOLEAN is_verified
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    hotels {
        UUID id PK
        UUID owner_id FK,UK "users.id (1 admin=1 hotel)"
        VARCHAR name
        TEXT description
        TEXT address
        VARCHAR city
        VARCHAR district
        DECIMAL latitude
        DECIMAL longitude
        VARCHAR phone
        VARCHAR email
        SMALLINT star_rating "CHECK 1-5"
        ENUM status "pending|approved|rejected|suspended"
        TEXT rejection_reason
        DECIMAL avg_rating "denormalized"
        INTEGER total_reviews "denormalized"
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    hotel_images {
        UUID id PK
        UUID hotel_id FK "hotels.id CASCADE"
        TEXT image_url
        BOOLEAN is_primary
        INTEGER sort_order
        TIMESTAMPTZ created_at
    }

    room_types {
        UUID id PK
        UUID hotel_id FK "hotels.id"
        VARCHAR name
        TEXT description
        DECIMAL base_price "CHECK gt 0"
        INTEGER max_guests "CHECK gt 0"
        DECIMAL area_sqm
        VARCHAR bed_type
        INTEGER total_rooms "CHECK gt 0"
        BOOLEAN is_active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    room_type_images {
        UUID id PK
        UUID room_type_id FK "room_types.id CASCADE"
        TEXT image_url
        BOOLEAN is_primary
        INTEGER sort_order
        TIMESTAMPTZ created_at
    }

    rooms {
        UUID id PK
        UUID room_type_id FK "room_types.id"
        VARCHAR room_number UK
        INTEGER floor
        ENUM status "available|occupied|cleaning|maintenance"
        BOOLEAN is_active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    amenities {
        UUID id PK
        VARCHAR name UK
        VARCHAR icon
        VARCHAR category
        TIMESTAMPTZ created_at
    }

    room_type_amenities {
        UUID room_type_id PK,FK "room_types.id"
        UUID amenity_id PK,FK "amenities.id"
    }

    bookings {
        UUID id PK
        VARCHAR booking_code UK
        UUID user_id FK "users.id"
        UUID hotel_id FK "hotels.id"
        DATE check_in_date
        DATE check_out_date "CHECK gt check_in"
        INTEGER num_guests "CHECK gt 0"
        DECIMAL total_room_price
        DECIMAL total_service_price
        DECIMAL discount_amount
        DECIMAL total_amount
        ENUM status "pending|confirmed|checked_in|checked_out|cancelled|no_show"
        TEXT cancellation_reason
        TIMESTAMPTZ cancelled_at
        UUID cancelled_by FK "users.id"
        UUID promotion_id FK "promotions.id"
        TEXT special_requests
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    booking_rooms {
        UUID id PK
        UUID booking_id FK "bookings.id CASCADE"
        UUID room_type_id FK "room_types.id"
        UUID room_id FK "rooms.id NULLABLE"
        INTEGER quantity "CHECK gt 0"
        DECIMAL price_per_night "snapshot"
        INTEGER num_nights "CHECK gt 0"
        DECIMAL subtotal
    }

    payments {
        UUID id PK
        UUID booking_id FK "bookings.id"
        DECIMAL amount
        ENUM payment_method "zalopay|momo|credit_card|bank_transfer"
        ENUM payment_status "pending|completed|failed|refunded"
        VARCHAR transaction_id
        JSONB payment_gateway_response
        TIMESTAMPTZ paid_at
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }

    invoices {
        UUID id PK
        VARCHAR invoice_number UK
        UUID booking_id FK,UK "bookings.id (1:1)"
        UUID payment_id FK "payments.id"
        UUID user_id FK "users.id"
        UUID hotel_id FK "hotels.id"
        DECIMAL total_room_price
        DECIMAL total_service_price
        DECIMAL discount_amount
        DECIMAL total_amount
        TIMESTAMPTZ issued_at
        TIMESTAMPTZ created_at
    }

    hotel_services {
        UUID id PK
        UUID hotel_id FK "hotels.id CASCADE"

    booking_services {

    room_status_logs {
        UUID id PK
        UUID room_id FK "rooms.id CASCADE"

        VARCHAR new_status

    reviews {
        UUID id PK
        UUID user_id FK "users.id"
        UUID hotel_id FK "hotels.id"
        UUID booking_id FK "bookings.id"

        TIMESTAMPTZ updated_at
    }

    favorites {

    }

    promotions {
        UUID id PK
        UUID hotel_id FK "hotels.id CASCADE"

        TEXT description
        ENUM discount_type "percentage|fixed_amount"
        DECIMAL discount_value "CHECK gt 0"
        DECIMAL min_booking_amount
        DECIMAL max_discount_amount
        DATE start_date
        DATE end_date "CHECK gte start"
        INTEGER usage_limit
        INTEGER used_count
        BOOLEAN is_active
        TIMESTAMPTZ created_at
        TIMESTAMPTZ updated_at
    }
